\begin{equation*} LHS=4Ae^{-t}+B=2e^{-t}+3=RHS \end{equation*}

hence

\begin{align*} 4A=2 \amp ,B=3\\ A=\frac{1}{2} \end{align*}

hence

\begin{equation*} y_{p}=\frac{1}{2}te^{-t}+3 \end{equation*}

so that the General Solution is

\begin{equation*} y=c_{1}e^{t}+c_{2}te^{t}+c_{3}e^{-t}+\frac{1}{2}te^{-t}+3. \end{equation*}
