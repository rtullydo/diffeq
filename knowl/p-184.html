\begin{align*} m\frac{dv}{dt}=\frac{v}{128}-mg \amp \iff2\frac{dv}{dt}=\frac{v}{128}-64\\ \amp \iff\frac{dv}{dt}=\frac{v}{256}-32\\ \amp \iff\int\frac{dv}{v-256\cdot32}=\int\frac{1}{256}dt\\ \amp \iff v(t)=Ce^{t/256}+256\cdot32. \end{align*}
