\begin{align*} x_{1}(t) \amp =\frac{1}{3}e^{t}+\frac{2}{3}e^{-2t}\\ x_{2}(t) \amp =\frac{1}{3}e^{t}-\frac{4}{3}e^{-2t} \end{align*}

\begin{equation*} \begin{cases} x_{1}^{\prime}=x_{2} \amp x_{1}(0)=1\\ x_{2}^{\prime}=2x_{1}-x_{2} \amp x_{2}(0)=-1 \end{cases}. \end{equation*}
